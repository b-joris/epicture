Future<bool> addComment(
  String imageID,
  String comment, {
  int parentID = -1,
}) async {
  try {
    final data = await _repository.addCommentData(
      imageID,
      comment,
      parentID: parentID,
    );
    if (data['success']) return true;
  } catch (_) {
    return false;
  }
  return false;
}
