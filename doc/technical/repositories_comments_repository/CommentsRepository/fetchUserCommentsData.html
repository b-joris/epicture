Future<List<Comment>> fetchUserCommentsData({
  String username = 'me',
  String commentSort = 'newest',
  int page = 0,
}) async {
  final data =
      await _provider.get('account/$username/comments/$commentSort/$page');
  final comments =
      List<Comment>.from(data['data'].map((data) => Comment.fromJson(data)));
  return comments;
}
