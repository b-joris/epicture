Future<List<Post>> fetchAccountPostsData({
  int page = 0,
}) async {
  final data = await _provider.get('account/me/images/$page');
  final posts =
      List<Post>.from(data['data'].map((data) => Post.fromJson(data)));
  return posts;
}
